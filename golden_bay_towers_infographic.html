        async function fetchGeminiResponse(userPrompt, systemInstruction, enableSearch = false) {

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${API_KEY}`;

            

            // Build the budget data string for context

            let contextData = "Golden Bay Towers 2026 Budget Data (USD):\n\n";

            contextData += "REVENUES:\n" + JSON.stringify(budgetData.revenues, null, 2) + "\n\n";

            contextData += "EXPENSES:\n" + JSON.stringify(budgetData.expenses, null, 2) + "\n\n";

            contextData += "AGGREGATED EXPENSES:\n" + JSON.stringify(expenseCategories, null, 2) + "\n\n";


            const fullPrompt = `CONTEXT: ${contextData}\n\nUSER QUERY: ${userPrompt}`;


            const payload = {

                contents: [{ parts: [{ text: fullPrompt }] }],

                systemInstruction: { parts: [{ text: systemInstruction }] },

                tools: enableSearch ? [{ "google_search": {} }] : undefined,

            };


            let response;

            let result;

            const maxRetries = 3;

            let delay = 1000;


            for (let i = 0; i < maxRetries; i++) {

                try {

                    response = await fetch(apiUrl, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify(payload)

                    });


                    result = await response.json();

                    

                    if (response.ok) {

                        break; // Success, exit loop

                    }


                } catch (error) {

                    console.error(`Attempt ${i + 1} failed:`, error);

                }


                if (i < maxRetries - 1) {

                    await new Promise(resolve => setTimeout(resolve, delay));

                    delay *= 2; // Exponential backoff

                }

            }


            if (!result || !result.candidates || result.candidates.length === 0) {

                return "Error: Could not get a response from the AI assistant.";

            }

            

            return result.candidates[0].content.parts[0].text;

        }


        async function handleQnA() {

            const inputElement = document.getElementById('qnaInput');

            const outputElement = document.getElementById('qnaOutput');

            const spinner = document.getElementById('qnaSpinner');

            const question = inputElement.value.trim();


            if (!question) {

                outputElement.innerHTML = "Please enter a question about the budget.";

                return;

            }


            // Show loading state

            spinner.style.display = 'inline-block';

            outputElement.innerHTML = "Analyzing budget data and generating response...";


            const systemInstruction = "You are a professional, transparent financial assistant for the Golden Bay Towers condo association. Your only role is to answer questions based *strictly* on the provided 2026 budget data. Use a clear, concise, and reassuring tone. Do not invent any numbers or categories. If a term is not in the data, state that you can only comment on the provided line items. Keep answers brief (1-3 sentences).";

            

            try {

                const answer = await fetchGeminiResponse(question, systemInstruction, false);

                outputElement.innerHTML = answer;

            } catch (error) {

                outputElement.innerHTML = "An unexpected error occurred. Please try again.";

                console.error("QnA API Error:", error);

            } finally {

                spinner.style.display = 'none';

            }

        }

        function setupModal() {

            const modal = document.getElementById("qnaModal");

            const btn = document.getElementById("openQnAModal");

            const span = document.getElementsByClassName("close-button")[0];

            const submitBtn = document.getElementById('submitQna');

            const qnaInput = document.getElementById('qnaInput');


            btn.onclick = function() {

                modal.style.display = "flex";

                qnaInput.focus();

            }


            span.onclick = function() {

                modal.style.display = "none";

            }


            window.onclick = function(event) {

                if (event.target == modal) {

                    modal.style.display = "none";

                }

            }

            

            submitBtn.onclick = handleQnA;


            qnaInput.addEventListener('keypress', function(e) {

                if (e.key === 'Enter') {

                    handleQnA();

                }

            });

        }


        window.onload = function() {

            setupChart();

            setupModal();
